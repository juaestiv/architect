Angular NGRX State management

  1. Install the ngrx package and the devtools for their use with chrome extensio
ng add @ngrx/store@latest
ng add @ngrx/store-devtools@latest
  1. Review changes in app.module.ts. There should be the import of the StoreModule and the entry in the import section. It will be needed to modifiy this section later on
import { StoreModule } from '@ngrx/store';
  imports: [
    BrowserModule,
    AppRoutingModule,
    StoreModule.forRoot({}, {})
  ],
  1. Create counter.actions.ts. The text string will be shown with the redux dev tools extension if properly installed. This action will be called from the component like this.store.dispatch(increment()); , the reduder will listen that action like on(increment, (state) => state + 1),
import { createAction } from '@ngrx/store';

export const increment = createAction('[Counter Component] Increment');
export const decrement = createAction('[Counter Component] Decrement');
export const reset = createAction('[Counter Component] Reset');
  1. counter.reducer.ts The reducer has to import the actions to be able to listen them and react.
import { createReducer, on } from '@ngrx/store';
import { increment, decrement, reset } from './counter.actions';

export const initialState = 0;

export const counterReducer = createReducer(
  initialState,
  on(increment, (state) => state + 1),
  on(decrement, (state) => state - 1),
  on(reset, (state) => 0)
);
  1. comeback to app.module.ts. The reducer must be imporeted and intialized here as follows
import { counterReducer } from './counter.reducer';
  imports: [
    BrowserModule,
    AppRoutingModule,
    StoreModule.forRoot({ count: counterReducer })
  ],
  1. app Component
import { Component } from '@angular/core';
import { Observable } from 'rxjs';
import { Store } from '@ngrx/store';
import { increment, decrement, reset } from './counter.actions';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent {
  count$: Observable<number>
  title = '101-angular-ngrx';
  constructor(private store: Store<{ count: number }>) {
    this.count$ = store.select('count');
  }

  increment() {
    this.store.dispatch(increment());
  }
 
  decrement() {
    this.store.dispatch(decrement());
  }
 
  reset() {
    this.store.dispatch(reset());
  }


}
  1. app template
<button (click)="increment()">Increment</button>
<div>Current Count: {{ count$ | async }}</div>
<button (click)="decrement()">Decrement</button>
<button (click)="reset()">Reset Counter</button>
