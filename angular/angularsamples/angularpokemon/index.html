Service

ng generate service pokemon

Edit the service (edit the http entries like below)

import { Injectable } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http'; // This
import { Observable } from 'rxjs'; // This
import { environment } from '../environments/environment'; // This
import { Pokemon } from './pokemon'; // This

@Injectable({
  providedIn: 'root'
})
export class PokemonService {

  constructor(private http: HttpClient) { } // This
  getPokemon(pokemon: string): Observable<Pokemon> { // This
    const options = new HttpParams()// This
    //.set('units', 'metric')// This
    return this.http.get<Pokemon>(environment.apiUrl + 'pokemon' + pokemon);// This
  }
}
