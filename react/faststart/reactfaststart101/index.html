props to components

send with spread operator

function App() {
  interface IProps  {
    name:string,
    age?:number
  }
  const propertySent:IProps = {
    name:"Hiya"
  }

  return (
   
   <>
   <Header {...propertySent} />
   </>
  );
}

type whatever = {
    name:string,
    age?:number

const Header = (propertyReceived:whatever) => {

  )
}
